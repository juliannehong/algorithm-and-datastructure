// sorting algorithms
#include <iostream>
#include <vector>
using namespace std;

void swap(int& x, int& y){
  int temp = x;
  x = y;
  y = temp;
}

void quick_sort(vector<int>& v, int l, int r){
  if (l == r)
    return ;
  else{
    int pivot = r, low = l, high = r - 1;
    while (low < high){
      if (v[low] >= v[pivot] && v[high] <= v[pivot]){
        swap(v[low], v[high]);
      }
      if (v[low] < v[pivot]){
        low++;
      }
      if (v[high] > v[pivot]){
        high--;
      }
    }
    swap(v[low], v[pivot]);
    quick_sort(v, l, low-1);
    quick_sort(v, low+1, r);
  }
}
