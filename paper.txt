#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

int main() {
	string s = "abbbbccddd";
	vector<int> v{ 1,4,7,9 };
	string re = "";

	for (int i = 0; i < s.size(); ++i) {
		auto iter = find(v.begin(), v.end(), i);
		if (iter != v.end()) {
			if (distance(v.begin(), iter) % 2 == 0) {
				re += "<b>";
				re += s[i];
			}
			else {
				re += s[i];
				re += "</b>";
			}
		}
		else {
			re += s[i];
		}
	}
	cout << re << endl;
	system("pause");
	return 0;
}
